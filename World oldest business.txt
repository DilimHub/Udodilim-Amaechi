BusinessFinancing.co.uk researched the oldest company still in business in almost every country and compiled the results into several CSV files. This dataset has been cleaned.

-- Which business categories are best suited to last over the course of centuries?
SELECT countries.continent,
categories.category,
businesses.year_founded
FROM businesses
INNER JOIN public.categories
USING (category_code)
INNER JOIN public.countries
USING (country_code)
GROUP BY categories.category,countries.continent,businesses.year_founded
ORDER BY businesses.year_founded

correct verison 
SELECT c.continent, cat.category, MIN(b.year_founded) AS year_founded
FROM businesses b
INNER JOIN categories cat ON b.category_code = cat.category_code
INNER JOIN countries c ON b.country_code = c.country_code
GROUP BY c.continent, cat.category
ORDER BY c.continent, cat.category;

-- How many countries per continent lack data on the oldest businesses
-- Does including the `new_businesses` data change this?
SELECT COUNT(countries.country) AS country_count,
       countries.continent
FROM countries
LEFT JOIN businesses 
ON countries.country_code = businesses.country_code
LEFT JOIN new_businesses 
ON countries.country_code = new_businesses.country_code
WHERE businesses.business IS NULL AND new_businesses.business IS NULL
GROUP BY countries.continent;

SELECT c.continent, COUNT(c.country_code) AS countries_without_businesses
FROM countries c
LEFT JOIN (
    SELECT * FROM businesses
    UNION ALL
    SELECT * FROM new_businesses
) b
ON c.country_code = b.country_code
WHERE b.business IS NULL
GROUP BY c.continent;

-- What is the oldest business on each continent?
SELECT business,
year_founded,
continent,
country
FROM businesses
INNER JOIN countries
USING (country_code)
ORDER BY year_founded

SELECT bc.continent, bc.business, bc.year_founded, bc.country
FROM (
    SELECT business, year_founded, country, continent
    FROM businesses
    JOIN countries USING (country_code)
) bc
JOIN (SELECT continent, MIN(year_founded) AS year_founded
    FROM businesses
    JOIN countries 
    USING (country_code)
    GROUP BY continent
) c 
ON bc.continent = c.continent AND bc.year_founded = c.year_founded
